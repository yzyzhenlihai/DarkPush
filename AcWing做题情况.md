```c++
struct Sum{
	int s,c,d;
	//重载<，用于结构体排序
	bool operator< (const Sum& t){
		if(s!=t.s)return s<t.s;
		if(c!=t.c)return c<t.c;
		return d<t.d;
	} 
}sum[N]; //用来存每一种组合情况 
```
